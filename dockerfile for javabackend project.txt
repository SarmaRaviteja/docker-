FROM alpine/git
WORKDIR /app
RUN git clone https://github.com/GOPURAJU/empjavaupdated.git



FROM maven:3.6.3-jdk-11-slim AS MAVEN_TOOL_CHAIN
WORKDIR /app
COPY --from=0 /app/empjavaupdated /app
RUN mvn install



FROM tomcat:9.0-alpine
MAINTAINER "admin"
USER root
COPY --from=MAVEN_TOOL_CHAIN /app/target/LESM-Status-Monitor-0.0.1-SNAPSHOT.war /usr/local/tomcat/webapps
EXPOSE 8080
CMD ["catalina.sh", "run"]
